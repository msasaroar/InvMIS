 	   Controllers/
		Program.cs
 		appsettings.json
 	InvMIS.Application/		=>[ Project : InvMIS.Application (Class Library) - Interfaces & Services ]
 	   Interfaces/
 	   Services/


 	   DTOs/
 	   Validation/
 	InvMIS.Domain/		=>[ Project : InvMIS.Domain (Class Library) - Entities ]
 	   Entities/
 	InvMIS.Infrastructure/	=>[ Project : InvMIS.Infrastructure (Class Library) - DbContext & Repositories	]
 	   Data/
 	   Repositories/
